83 void set_module(Modules::Type type, std::string const& name)
84 {
85 const auto set = [&](auto& module) {
86 auto [mod, config] =
87 load_module<decltype(*module.get())>(name, _installDir, type);
88 module = std::move(mod);
89 _configs[type] = std::move(config);
90 };
91
92 if ( type == Modules::Type::Output ) {
93 auto [mod, config] =
94 load_module<Modules::Output>(name, _installDir, type);
95 _output.add_output(std::move(mod), std::move(config));
96
97 return;
98 }
99
100 switch ( type ) {
101 case Modules::Type::Model:
102 set(_model);
103 break;
104 case Modules::Type::Goals:
105 set(_goals);
106 break;
107 case Modules::Type::Pedestrians:
108 set(_pedset);
109 break;
110 case Modules::Type::Map:
111 set(_map);
112 break;
113 case Modules::Type::PedInput:
114 if ( name == "JSON" ) {
115 _pedInput = std::make_unique<VIPRA::Input::JSON>();
116 _configs[type] = [](void*, Parameters&, VIPRA::Random::Engine&) {};
117 break;
118 }
119 set(_pedInput);
120 break;
121 case Modules::Type::MapInput:
122 if ( name == "JSON" ) {
123 _mapInput = std::make_unique<VIPRA::Input::JSON>();
124 _configs[type] = [](void*, Parameters&, VIPRA::Random::Engine&) {};
125 break;
126 }
127 set(_mapInput);
128 break;
129 case Modules::Type::Output:
130 case Modules::Type::Parameters:
131 case Modules::Type::Simulation:
132 case Modules::Type::Behavior_model:
133 throw std::runtime_error("Can't Load " + to_string(type) + " Modules");
134 }
135 }

193inline void Simulation::run_sim(Parameters& params)
194{
195 params.reset();
196
197 initialize(params);
198
199 _output.new_run(_currSimIdx++);
200
201 _currTimestep = 0;
202
203 VIPRA::State state;
204 state.initialize(*_pedset);
205
206 // main loop
207 while ( _currTimestep < _maxTimestep && ! _goals->is_sim_goal_met() ) {
208 _model->timestep(*_pedset, *_map, *_goals, state, _timestepSize,
209 _currTimestep);
210 _behaviorModel.timestep(*_pedset, *_map, *_goals, state, _timestepSize);
211 _pedset->update(state);
212 _goals->update(*_pedset, *_map, _timestepSize);
213
214 if ( _currTimestep % _outputFrequency == 0 )
215 _output.timestep_update(_currTimestep, _timestepSize, state);
216
217 ++_currTimestep;
218 }
219
220 if ( _outputParams )
221 _output.write_to_file("parameters.json", params.get_used_parameters());
222
223 _output.write();
224}

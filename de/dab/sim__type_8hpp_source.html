148template <typename model_t, typename output_t, typename pedset_t, typename goals_t, typename obstacles_t>
149template <typename pedinput_t, typename obsinput_t, typename params_t>
150auto SimType<model_t, output_t, pedset_t, goals_t, obstacles_t>::run_sim(pedinput_t const& pedInput,
151 obsinput_t const& obsInput,
152 params_t&& params) -> output_data_t
153{
154 // initialize parameters, etc
155 register_step(std::forward<params_t>(params));
156
157 initialize(pedInput, obsInput, std::forward<params_t>(params));
158
159 _output.new_run(_currSimIdx++);
160
161 _currTimestep = 0;
162
163 VIPRA::State state;
164 state.initialize(_pedset);
165
166 // main loop
167 while ( _currTimestep < _maxTimestep && ! _goals.is_sim_goal_met() ) {
168 _model.timestep(_pedset, _map, _goals, _output, state, _timestepSize, _currTimestep);
169 _behaviorModel.timestep(_pedset, _map, _goals, state, _timestepSize);
170 _pedset.update(state);
171 _goals.update(_pedset, _map, _timestepSize);
172 output_positions();
173 ++_currTimestep;
174 }
175
176 // write output and if output returns a value, return that
177 if constexpr ( std::is_same_v<output_data_t, void> ) {
178 _output.write();
179 }
180 else {
181 return _output.write();
182 }
183}

238template <typename model_t, typename output_t, typename pedset_t, typename goals_t, typename obstacles_t>
239void SimType<model_t, output_t, pedset_t, goals_t, obstacles_t>::output_positions()
240{
241 if ( _currTimestep % _outputFrequency != 0 ) {
242 return;
243 }
244
245 const VIPRA::size pedCnt = _pedset.num_pedestrians();
246 auto const& coords = _pedset.all_coords();
247
248 for ( VIPRA::idx i = 0; i < pedCnt; ++i ) {
249 _output.ped_timestep_value(i, _currTimestep / _outputFrequency, "position", coords.at(i));
250 }
251}

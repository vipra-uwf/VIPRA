104 // TODO(rolland): issue #25 this assumes that a only a single node should ever run this function, there may be sitations where this isn't user friendly?

139template <typename model_t, typename output_t, typename pedset_t, typename goals_t, typename map_t>
140template <typename pedinput_t, typename obsinput_t, typename params_t>
141auto SimType<model_t, output_t, pedset_t, goals_t, map_t>::run_sim(pedinput_t const& pedInput,
142 obsinput_t const& obsInput,
143 params_t&& params) -> output_data_t
144{
145 params.reset();
146
147 // initialize parameters, etc
148 register_step(std::forward<params_t>(params));
149
150 initialize(pedInput, obsInput, std::forward<params_t>(params));
151
152 _output.new_run(_currSimIdx++);
153
154 _currTimestep = 0;
155
156 VIPRA::State state;
157 state.initialize(_pedset);
158
159 // main loop
160 while ( _currTimestep < _maxTimestep && ! _goals.is_sim_goal_met() ) {
161 _model.timestep(_pedset, _map, _goals, state, _timestepSize, _currTimestep);
162 _behaviorModel.timestep(_pedset, _map, _goals, state, _timestepSize);
163 _pedset.update(state);
164 _goals.update(_pedset, _map, _timestepSize);
165 output_positions();
166 ++_currTimestep;
167 }
168
169 if ( _outputParams ) _output.write_to_file("parameters.json", params.get_used_parameters());
170
171 // write output and if output returns a value, return that
172 if constexpr ( std::is_same_v<output_data_t, void> ) {
173 _output.write();
174 }
175 else {
176 return _output.write();
177 }
178}

233template <typename model_t, typename output_t, typename pedset_t, typename goals_t, typename map_t>
234void SimType<model_t, output_t, pedset_t, goals_t, map_t>::output_positions()
235{
236 if ( _currTimestep % _outputFrequency != 0 ) {
237 return;
238 }
239
240 const VIPRA::size pedCnt = _pedset.num_pedestrians();
241 auto const& coords = _pedset.all_coords();
242
243 for ( VIPRA::idx i = 0; i < pedCnt; ++i ) {
244 _output.ped_timestep_value(i, _currTimestep / _outputFrequency, "position", coords.at(i));
245 }
246}

98 void set_end_goals(auto const& pedset, auto const& map,
99 VIPRA::Random::Engine& engine)
100 {
101 assert(pedset.num_pedestrians() > 0);
102
103 const VIPRA::size pedCnt = pedset.num_pedestrians();
104
105 // find the end goals, provided as a module parameter
106 auto const& objects = map.get_objectives(_endGoalType);
107 if ( objects.empty() ) {
108 throw std::runtime_error("No objects of type " + _endGoalType +
109 " found in map");
110 }
111
112 // set each pedestrians end goal as the nearest of that type
113 for ( VIPRA::idx pedIdx = 0; pedIdx < pedCnt; ++pedIdx ) {
114 auto const pos = pedset.ped_coords(pedIdx);
115
116 auto const nearestGoalIter = get_nearest_goal(pos, objects);
117 VIPRA::Log::debug("Ped {} Nearest End Goal: ({}, {})", pedIdx,
118 (*nearestGoalIter).center().x,
119 (*nearestGoalIter).center().y);
120
121 if ( nearestGoalIter != objects.end() ) {
122 const auto point = (*nearestGoalIter).random_point(engine);
123 VIPRA::Log::debug("Ped {} Final End Goal: ({}, {})", pedIdx, point.x,
124 point.y);
125 set_end_goal(pedIdx, point);
126 }
127 else {
128 throw std::runtime_error("No goal found for pedestrian");
129 }
130 }

139 [[nodiscard]] static auto squash_path(std::vector<VIPRA::f3d> const& path,
140 VIPRA::Random::Engine& /*engine*/)
141 -> std::vector<VIPRA::f3d>
142 {
143 assert(path.empty() == false);
144
145 std::vector<VIPRA::f3d> squashedPath;
146
147 VIPRA::f3d dif;
148
149 for ( VIPRA::idx i = 1; i < path.size(); ++i ) {
150 auto currDif = path[i] - path[i - 1];
151 if ( currDif != dif ) {
152 squashedPath.push_back(path[i - 1]);
153 dif = currDif;
154 }
155 }
156
157 return squashedPath;
158 }

167 [[nodiscard]] static auto get_nearest_goal(
168 VIPRA::f3d pos, std::vector<VIPRA::Geometry::Polygon> const& goals)
169 -> std::vector<VIPRA::Geometry::Polygon>::const_iterator
170 {
171 assert(goals.empty() == false);
172
173 return std::min_element(goals.begin(), goals.end(),
174 [&](auto const& left, auto const& right) {
175 return pos.distance_to(left.center()) <
176 pos.distance_to(right.center());
177 });
178 }

186 void find_path(VIPRA::idx pedIdx, VIPRA::f3d startPos,
187 VIPRA::Random::Engine& engine)
188 {
189 // Get the pedestrian start and end grid location
190 VIPRA::idx startIdx = _graph.get_closest_grid_idx(startPos);
191 VIPRA::idx endIdx = _graph.get_closest_grid_idx(end_goal(pedIdx));
192
193 if ( _graph.nodes().size() <= startIdx ||
194 _graph.nodes().size() <= endIdx ) {
195 throw std::runtime_error("Start or end index is out of bounds");
196 }
197
198 // use A* to find the path
199 auto const path = VIPRA::Algo::astar(
200 startIdx, endIdx, _graph,
201 [&](VIPRA::idx left, VIPRA::idx right) -> VIPRA::f_pnt {
202 // distance comparison function
203 return _graph.data(left).pos.distance_to(_graph.data(right).pos);
204 },
205 [&](VIPRA::idx nodeIdx) -> VIPRA::f3d {
206 return _graph.data(nodeIdx).pos;
207 });
208
209 if ( ! path ) {
210 // No path found
211 throw std::runtime_error("No path found for pedestrian " +
212 std::to_string(pedIdx));
213 }
214
215 // set their path, squash nodes that all go in the same direction into one node
216 _paths[pedIdx] = squash_path(*path, engine);
217 }

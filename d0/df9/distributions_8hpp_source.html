23// NOLINTNEXTLINE (rolland) sticking to familiar naming convention for std::uniform_real_distribution
24class uniform_distribution {
25 public:
26 constexpr uniform_distribution(data_t min, data_t max) : _range(min, max) {}
27 constexpr explicit uniform_distribution(std::pair<data_t, data_t> rangeVals) : _range(std::move(rangeVals))
28 {
29 }
30
31 constexpr auto operator()(Random::Engine& engine) -> data_t
32 {
33 auto val = engine();
34 if ( (_range.second - _range.first) == 0 ) return _range.first;
35 return static_cast<data_t>(_range.first + static_cast<double>(val) /
36 (static_cast<double>(std::numeric_limits<uint64_t>::max()) /
37 static_cast<double>(_range.second - _range.first)));
38 }
39
40 [[nodiscard]] auto min() -> data_t { return _range.first; }
41 [[nodiscard]] auto max() -> data_t { return _range.second; }
42
43 private:
44 std::pair<data_t, data_t> _range{};
45};

52// NOLINTNEXTLINE (rolland) sticking to familiar naming convention for std::normal_distribution
53class normal_distribution {
54 // TODO(rolland): issue #24 this is rather slow
55
56 public:
57 constexpr normal_distribution(data_t mean, data_t stdDev) : _mean(mean), _stdDev(stdDev) {}
58
59 constexpr auto operator()(Random::Engine& engine) -> data_t
60 {
61 double resultX = 0;
62 double resultY = 0;
63 double result = 1.0;
64
65 if ( _hasSpare ) {
66 _hasSpare = false;
67 return static_cast<data_t>(_mean + _stdDev * _spare);
68 }
69
70 while ( result >= 1.0 || result == 0.0 ) {
71 resultX = ((static_cast<double>(engine()) / static_cast<double>(Random::Engine::max())) * TWO) - 1.0;
72 resultY = ((static_cast<double>(engine()) / static_cast<double>(Random::Engine::max())) * TWO) - 1.0;
73 result = resultX * resultX + resultY * resultY;
74 };
75
76 result = std::sqrt(-TWO * std::log(result) / result);
77 _spare = resultY * result;
78 _hasSpare = true;
79 return static_cast<data_t>(_mean + _stdDev * resultX * result);
80 }
81
82 private:
83 data_t _mean;
84 data_t _stdDev;
85
86 double _spare{0.0F};
87 bool _hasSpare = false;
88
89 static constexpr VIPRA::f_pnt TWO = 2.0;
90};

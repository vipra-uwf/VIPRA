95void Model::calm::calc_neighbors(auto const& pedset, auto const& /*map*/, auto const& goals)
96{
97 const VIPRA::size pedCnt = pedset.num_pedestrians();
98 auto const& coords = pedset.all_coords();
99
100 for ( VIPRA::idx i = 0; i < pedCnt; ++i ) {
101 if ( goals.is_goal_met(i) ) continue;
102
103 VIPRA::f3d pedCoords = coords[i];
104 VIPRA::f3d pedGoal = goals.current_goal(i);
105 VIPRA::Geometry::Rectangle pedRect = rect_from_shoulders(i, pedCoords, pedGoal);
106
107 auto nearest = pedset.closest_ped(i, [&](VIPRA::idx other) VIPRA_PERF_FUNC {
108 if ( i == other || goals.is_goal_met(other) ) return false;
109
110 auto otherCoords = coords[other];
111 VIPRA::f_pnt distance = pedCoords.distance_to(otherCoords);
112
113 if ( (_collision.raceStatuses[i] == 0 && _collision.raceStatuses[other] == 1) ||
114 std::fabs(distance - EQUILIBRIUM_DISTANCE) < EQUILIBRIUM_RESOLUTION )
115 return false;
116
117 if ( ! is_ped_toward_goal(pedCoords, pedGoal, otherCoords) ) return false;
118
119 if ( ! obj_spatial_test(pedRect, _peds.shoulders[other].start, _peds.shoulders[other].end) )
120 return false;
121
122 return true;
123 });
124
125 if ( nearest == i ) {
126 _peds.nearestDists[i] = std::numeric_limits<VIPRA::f_pnt>::max();
127 }
128 else {
129 _peds.nearestDists[i] = pedCoords.distance_to(coords[nearest]);
130 }
131 }

134void Model::calm::update_state(auto const& pedset, auto const& goals, VIPRA::State& state,
135 VIPRA::delta_t deltaT)
136{
137 const VIPRA::size pedCnt = pedset.num_pedestrians();
138 auto const& velocities = pedset.all_velocities();
139 auto const& coords = pedset.all_coords();
140
141 for ( VIPRA::idx i = 0; i < pedCnt; ++i ) {
142 if ( goals.is_goal_met(i) || _collision.status(i) == CALM::WAIT ) {
143 state.velocities[i] = VIPRA::f3d{};
144 continue;
145 }
146 if ( goals.time_since_last_goal(i) > 0 && goals.time_since_last_goal(i) <= SLIDING_GOAL_TIME ) {
147 state.velocities[i].x = 0;
148 state.velocities[i].y = 0;
149 continue;
150 }
151
152 const VIPRA::f3d goal = goals.current_goal(i);
153 const VIPRA::f3d velocity = velocities[i];
154 const VIPRA::f3d coord = coords[i];
155 const VIPRA::f_pnt desiredSpeed = _peds.maxSpeeds[i];
156 const VIPRA::f_pnt mass = _peds.masses[i];
157 const VIPRA::f_pnt reactionTime = _peds.reactionTimes[i];
158 const VIPRA::f_pnt beta = _peds.betas[i];
159 const VIPRA::f3d direction = goal - coord;
160
161 const VIPRA::f3d propulsion = ((direction.unit() * desiredSpeed * beta - velocity) * mass) / reactionTime;
162
163 state.velocities[i] = ((propulsion / mass) * deltaT) + velocity;
164 state.positions[i] = coord + (state.velocities[i] * deltaT);
165 }
166}

168auto Model::calm::is_path_blocked(VIPRA::idx pedIdx, VIPRA::f3d velocity, VIPRA::f_pnt maxDist,
169 auto const& map) -> VIPRA::f_pnt
170{
171 VIPRA::Geometry::Line shoulders = _peds.shoulders[pedIdx];
172 if ( shoulders.start == shoulders.end ) {
173 return -1;
174 }
175
176 VIPRA::f_pnt const leftDist = map.ray_hit(shoulders.start, shoulders.start + (velocity.unit() * maxDist));
177 VIPRA::f_pnt const rightDist = map.ray_hit(shoulders.end, shoulders.end + (velocity.unit() * maxDist));
178
179 if ( leftDist == -1 ) {
180 return rightDist;
181 }
182 if ( rightDist == -1 ) {
183 return leftDist;
184 }
185
186 return std::min(leftDist, rightDist);
187}

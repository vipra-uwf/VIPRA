336inline auto DrwReader::get_arc_as_polygon(DRW_Coord center, double radius, double start_angle,
337 double end_angle) -> std::vector<VIPRA::f3d>
338{
339 std::vector<VIPRA::f3d> points;
340 double center_x = center.x;
341 double center_y = center.y;
342
343 // Increments by steps to create points around a center at the specified radius.
344 // For a circle, start point would be 0.00 and end point would be 2*pi
345 // Steps are defined as the angle resulting from subdividing 2*pi
346 double step_size = (2 * M_PI) / NUM_SUBDIVISIONS;
347 // TODO: consider tolerance with end state. Could create two points very close together.
348 for ( double step = start_angle; step < end_angle; step += step_size ) {
349 double x = center_x + radius * cos(step);
350 double y = center_y + radius * sin(step);
351
352 points.emplace_back(x, y, 0);
353 }
354
355 // Should add point on last point if not circle. This closes
356 // the shape if the arc is attached to another shape
357 if ( std::abs(end_angle - (2 * M_PI)) > 0.01 ) {
358 points.emplace_back(center_x + (radius * cos(end_angle)), center_y + (radius * sin(end_angle)), 0);
359 }
360
361 return points;
362}

371inline void DrwReader::add_obstacle(std::vector<VIPRA::f3d> const& points, std::string object_type)
372{
373 transform(object_type.begin(), object_type.end(), object_type.begin(), ::toupper);
374
375 VIPRA::Geometry::Polygon shape(points);
376
377 if ( object_type == "OBSTACLES" ) {
378 _obstacles.emplace_back(shape);
379 }
380 else if ( object_type == "PEDESTRIANS" ) {
381 // _spawns.insert(_spawns.end(), points.begin(), points.end());
382 _spawns.emplace_back(shape);
383 }
384 else if ( object_type.find("AREAS") != std::string::npos ) {
385 _areas[object_type] = shape;
386 }
387 else {
388 if ( _objectives[object_type].empty() ) {
389 _objectives[object_type] = {shape};
390 }
391 else {
392 _objectives[object_type].emplace_back(shape);
393 }

54 [[nodiscard]] virtual auto ray_hit(VIPRA::f3d start,
55 VIPRA::f3d end) const -> VIPRA::f_pnt
56 {
57 VIPRA::f_pnt hit = std::numeric_limits<VIPRA::f_pnt>::max();
58 for ( auto const& obstacle : _obstacles ) {
59 for ( auto const& edge : obstacle.sides() ) {
60 if ( ! edge.does_intersect({start, end}) ) continue;
61 auto intersection = edge.intersection_point({start, end});
62 hit = std::min(hit, start.distance_to(intersection));
63 }
64 }
65 return hit;
66 }

192 auto setToMax = [&](auto const& polygon) {

195 _dimensions.y = std::max({_dimensions.y, edge.start.y, edge.end.y});

198 };
199
200 auto setToAllMax = [&](auto const& polygons) {
201 for ( auto const& polygon : polygons ) {
202 setToMax(polygon);
203 }
204 };
205
206 setToAllMax(obstacles);
207 setToAllMax(spawns);
208
209 for ( auto const& [_, polygons] : objectives ) {

218 _dimensions.y, _dimensions.z);

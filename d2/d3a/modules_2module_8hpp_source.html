55template <typename class_t>
56void Module<class_t>::register_params(Parameters& paramIn)
57{
58 static_assert(Concepts::has_parameters<class_t>,
59 "Module is missing VIPRA_REGISTER_PARAMS");
60
61 VIPRA::Log::debug("Registering Params For: {}", self().module_name());
62
63 // get module parameters customization point
64 auto params = self().module_params();
65
66 if constexpr ( ! std::is_same_v<std::tuple<>,
67 std::remove_cvref_t<decltype(params)>> ) {
68 // register all of the parameters with the parameter loader
69 std::apply(
70 [&](auto const& first, auto const&... restArgs) {
71 auto regis = [&](auto const& param) {
72 paramIn.register_param(self().module_type(), self().module_name(),
73 param.first);
74 };
75
76 // recursively register each parameter
77 regis(first);
78 (regis(restArgs), ...);
79 },
80 params);
81 }
82
83 VIPRA::Log::debug("Done Registering Params For: {}", self().module_name());
84}

93template <typename class_t>
94void Module<class_t>::config(Parameters& paramIn, VIPRA::Random::Engine& engine)
95{
96 static_assert(Concepts::has_parameters<class_t>,
97 "Module is missing VIPRA_REGISTER_PARAMS");
98
99 VIPRA::Log::debug("Configuring Module: {}", self().module_name());
100
101 // get module parameters customization point
102 auto const& params = self().module_params();
103
104 if constexpr ( ! std::is_same_v<std::tuple<>,
105 std::remove_cvref_t<decltype(params)>> ) {
106 // load each parameter and apply it to the coresponding variable
107 std::apply(
108 [&](auto const& first, auto const&... restArgs) {
109 auto configure = [&](auto const& param) {
110 using param_t = std::remove_cvref_t<decltype(param.second)>;
111 VIPRA::Log::debug("Loading Param: {}, {}", param.first,
112 static_cast<void*>(&param.second));
113
114 // set the parameter variable as loaded from input
115 param.second = paramIn.get_param<param_t>(self().module_type(),
116 self().module_name(),
117 param.first, engine);
118 };
119
120 // recursively load the parameters
121 configure(first);
122 (configure(restArgs), ...);
123 },
124 params);
125 }

40template <typename class_t>
41void Module<class_t>::register_params(Parameters& paramIn)
42{
43 static_assert(Concepts::has_parameters<class_t>,
44 "Module is missing VIPRA_REGISTER_PARAMS");
45
46 VIPRA::Log::debug("Registering Params For: {}", self().module_name());
47
48 // get module parameters customization point
49 auto params = self().module_params();
50
51 if constexpr ( ! std::is_same_v<std::tuple<>,
52 std::remove_cvref_t<decltype(params)>> ) {
53 // register all of the parameters with the parameter loader
54 std::apply(
55 [&](auto const& first, auto const&... restArgs) {
56 auto regis = [&](auto const& param) {
57 paramIn.register_param(self().module_type(), self().module_name(),
58 param.first);
59 };
60
61 // recursively register each parameter
62 regis(first);
63 (regis(restArgs), ...);
64 },
65 params);
66 }
67
68 VIPRA::Log::debug("Done Registering Params For: {}", self().module_name());
69}

78template <typename class_t>
79void Module<class_t>::config(Parameters& paramIn, VIPRA::Random::Engine& engine)
80{
81 static_assert(Concepts::has_parameters<class_t>,
82 "Module is missing VIPRA_REGISTER_PARAMS");
83
84 VIPRA::Log::debug("Configuring Module: {}", self().module_name());
85
86 // get module parameters customization point
87 auto const& params = self().module_params();
88
89 if constexpr ( ! std::is_same_v<std::tuple<>,
90 std::remove_cvref_t<decltype(params)>> ) {
91 // load each parameter and apply it to the coresponding variable
92 std::apply(
93 [&](auto const& first, auto const&... restArgs) {
94 auto configure = [&](auto const& param) {
95 using param_t = std::remove_cvref_t<decltype(param.second)>;
96 VIPRA::Log::debug("Loading Param: {}, {}", param.first,
97 static_cast<void*>(&param.second));
98
99 // set the parameter variable as loaded from input
100 param.second = paramIn.get_param<param_t>(self().module_type(),
101 self().module_name(),
102 param.first, engine);
103 };
104
105 // recursively load the parameters
106 configure(first);
107 (configure(restArgs), ...);
108 },
109 params);

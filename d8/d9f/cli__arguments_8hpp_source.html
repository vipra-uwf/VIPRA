56 static void parse(int argc, const char* const* argv)
57 {
58 get_args().clear();
59 auto args = format(argc, argv);
60
61 for ( auto arg = std::next(args.begin()); arg != args.end(); ++arg ) {
62 if ( arg->length() < 2 ) throw std::runtime_error("Unknown Flag: " + *arg);
63 if ( (*arg)[0] != '-' ) throw std::runtime_error("Unknown Flag: " + *arg);
64
65 auto [flag, val] = split_arg(*arg);
66
67 auto [fullValid, fullError] = validate_flag(flag, val);
68 if ( fullValid ) {

70 continue;
71 }
72
73 auto [compositeValid, compositeError] = validate_composite_flag(flag);
74 if ( compositeValid ) {
75 continue;
76 }

78 throw std::runtime_error(fullError + ": " += flag);
79 }

183 static auto split_arg(std::string_view arg) -> std::pair<std::string, std::string>
184 {
185 auto loc = arg.find('=');

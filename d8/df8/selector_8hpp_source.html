87template <typename select_t, typename context_t>
88void Selector<select_t, context_t>::run_selectors(const VIPRA::idxVec& selectorIdxs,
89 std::string const& behaviorName, auto pack)
90{
91 std::for_each(selectorIdxs.begin(), selectorIdxs.end(), [&](VIPRA::idx index) {
92 auto& selector = _subSelectors[index];
93 auto selectedPeds = select_peds_from_group(selector, pack, behaviorName);
94 update_ped_groups(selectedPeds, selector, pack.context, behaviorName);
95 });
96
97 _pedGroups[0].resize(pack.pedset.num_pedestrians());
98 std::iota(_pedGroups[0].begin(), _pedGroups[0].end(), 0);
99}

112template <typename select_t, typename context_t>
113auto Selector<select_t, context_t>::select_peds_from_group(select_t& selector, auto pack,
114 std::string const& behaviorName) -> VIPRA::idxVec
115{
116 auto const& fullGroup = _pedGroups.get_group(selector.group);
117 auto usablegroup = filter_used_peds(fullGroup, _pedGroups.get_used(selector.group));
118 auto result = selector.select_peds(fullGroup, usablegroup, pack);
119
120 if ( ! result.starved ) {
121 return result.group;
122 }
123
124 if ( selector.required ) {
125 VIPRA::Log::error("Behavior: {}, Required Selector Starved For Type: {} From Group: {}", behaviorName,
126 selector.type.fullType, selector.group);
127 DSLException::error();
128 }
129
130 VIPRA::Log::debug("Behavior: {}, Starved Selector For Type: {} From Group: {}", behaviorName,
131 selector.type.fullType, selector.group);
132 return result.group;
133}

146{
147 std::for_each(selectedPeds.begin(), selectedPeds.end(), [&](auto& pedIdx) {

149 VIPRA::Log::debug("Behavior: {}, Selecting Ped {} for Type: {}", behaviorName, pedIdx, type);
150 context.types[pedIdx] += type;
151 _pedGroups.add_ped(pedIdx, type);
152 });
153
154 _pedGroups.set_used(pedIdx, selector.group);
155 });

217template <typename select_t, typename context_t>
218auto Selector<select_t, context_t>::order_selectors() -> VIPRA::idxVec
219{
220 VIPRA::idxVec order;
221
222 for ( VIPRA::idx selIdx = 0; selIdx < _subSelectors.size(); ++selIdx ) {
223 if ( _subSelectors[selIdx].group == 0 ) {
224 order.push_back(selIdx);
225 }
226 }
227
228 _allTypes.for_each_type([&](typeUID type) {
229 for ( VIPRA::idx selIdx = 0; selIdx < _subSelectors.size(); ++selIdx ) {
230 if ( _subSelectors[selIdx].group == type ) {
231 order.push_back(selIdx);
232 }
233 }
234 });
235
236 for ( VIPRA::idx selIdx = 0; selIdx < _subSelectors.size(); ++selIdx ) {
237 bool notInGraph = std::find_if(order.begin(), order.end(),
238 [&](VIPRA::idx index) { return index == selIdx; }) == order.end();
239 if ( notInGraph ) {
240 order.push_back(selIdx);
241 }
242 }
243
244 check_for_duplicates(order);
245 return order;
246}

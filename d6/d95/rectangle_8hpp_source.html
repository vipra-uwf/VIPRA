108VIPRA_POLY_FUNC auto Rectangle::is_point_inside(f3d point) const noexcept
109 -> bool
110{
111 // NOLINTBEGIN(readability-identifier-length)
112 const f3d ab = _points[1] - _points[0];
113 const f3d am = point - _points[0];
114 const f3d bc = _points[2] - _points[1];
115 const f3d bm = point - _points[1];
116
117 return 0 <= ab.dot(am) && ab.dot(am) <= ab.dot(ab) && 0 <= bc.dot(bm) &&
118 bc.dot(bm) <= bc.dot(bc);

152VIPRA_POLY_FUNC auto Rectangle::rotation() const noexcept -> f_pnt
153{
154 constexpr f_pnt HALF = 180.0;
155 constexpr f_pnt FULL = 360.0;
156
157 const f_pnt xDif = _points[0].x - _points[2].x;
158 const f_pnt yDif = _points[0].y - _points[2].y;
159
160 f_pnt rotation = std::atan(xDif / yDif) * HALF / std::numbers::pi;
161 if ( yDif < 0 ) {
162 rotation += HALF;
163 }
164 else if ( xDif < 0 ) {
165 rotation += FULL;
166 }
167
168 return rotation;
169}

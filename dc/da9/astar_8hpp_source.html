95 struct PQueue : public std::priority_queue<Node*, std::vector<Node*>, typename Node::Compare> {
96 auto search(VIPRA::idx nodeIdx) -> Node*
97 {
98 auto container = this->c;
99 auto gridPoint =
100 std::find_if(container.begin(), container.end(), [&](Node* node) { return node->self == nodeIdx; });
101 if ( gridPoint == container.end() ) {
102 return nullptr;
103 }
104
105 return *gridPoint;
106 }
107 auto begin() -> decltype(this->c.begin()) { return this->c.begin(); }
108 auto end() -> decltype(this->c.end()) { return this->c.end(); }
109 };
110
111 std::vector<Node> nodes{graph.nodes().size()};
112 PQueue openset;
113 std::unordered_set<Node*> closedset;
114
115 nodes[start] = Node{start, start, 0, distance_func(start, end)};
116
117 Node* current = nullptr;
118 openset.push(&nodes[start]);
119
120 while ( ! openset.empty() ) {
121 current = openset.top();
122
123 if ( current->self == end ) {
124 break;
125 }
126
127 openset.pop();
128 closedset.insert(current);
129 for ( VIPRA::idx neighborIdx : graph.neighbors(current->self) ) {
130 if ( ! closedset.contains(&nodes[neighborIdx]) ) {
131 Node neighbor;
132 neighbor.self = neighborIdx;
133 neighbor.parent = current->self;
134 neighbor.distanceFromStart = current->distanceFromStart + distance_func(current->self, neighborIdx);
135 neighbor.distanceWithHeuristic = neighbor.distanceFromStart + distance_func(neighborIdx, end);
136
137 auto found = std::find(openset.begin(), openset.end(), &nodes[neighborIdx]);
138 if ( found == openset.end() ) {
139 nodes[neighborIdx] = neighbor;
140 openset.push(&nodes[neighborIdx]);
141 }
142 else {
143 if ( neighbor.distanceFromStart < (*found)->distanceFromStart ) {
144 (*found)->distanceFromStart = neighbor.distanceFromStart;
145 (*found)->distanceWithHeuristic = neighbor.distanceWithHeuristic;
146 (*found)->parent = neighbor.parent;
147 }
148 }
149 }
150 }
151 }
152
153 if ( current->self != end ) {
154 // no path found
155 return std::nullopt;
156 }
157
158 // construct the full path, If a conversion function is provided, run the path through that
159 ret_t path;
160 while ( current->self != start ) {
161 if constexpr ( std::is_same_v<conversion_f_t, VOID> ) {
162 path.push_back(current->self);
163 }
164 else {
165 path.push_back(conversion_func(current->self));
166 }
167
168 current = &nodes[current->parent];
169 }
170
171 // add the starting node
172 if constexpr ( std::is_same_v<conversion_f_t, VOID> ) {
173 path.push_back(start);
174 }
175 else {
176 path.push_back(conversion_func(start));
177 }
178
179 return path;
180}

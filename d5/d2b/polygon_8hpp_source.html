98constexpr auto Polygon::is_point_inside(f3d point) const noexcept -> bool
99{
100 // TODO(rolland): verify this
101 bool isInside = false;
102 for ( auto const& side : sides() ) {
103 if ( side.is_point_on(point) ) return true;
104 if ( side.start.y > point.y != side.end.y > point.y &&
105 point.x < (side.end.x - side.start.x) * (point.y - side.start.y) /
106 (side.end.y - side.start.y) +
107 side.start.x ) {
108 isInside = ! isInside;
109 }
110 }

146VIPRA_POLY_FUNC auto Polygon::bounding_box() const noexcept -> Rectangle
147{
148 f3d botLeft{std::numeric_limits<VIPRA::f_pnt>::max(),
149 std::numeric_limits<VIPRA::f_pnt>::max()};
150 f3d topRight{std::numeric_limits<VIPRA::f_pnt>::min(),
151 std::numeric_limits<VIPRA::f_pnt>::min()};
152 for ( auto const& point : _points ) {
153 topRight.x = std::max({topRight.x, point.x});
154 topRight.y = std::max({topRight.y, point.y});
155 botLeft.x = std::min({botLeft.x, point.x});
156 botLeft.y = std::min({botLeft.y, point.y});
157 }
158
159 f3d topLeft = f3d{botLeft.x, topRight.y};
160 f3d botRight = f3d{topRight.x, botLeft.y};
161
162 return Rectangle{botLeft, topLeft, topRight, botRight};
163}

165VIPRA_POLY_FUNC auto Polygon::sides() const noexcept -> std::vector<Line>
166{
167 std::vector<Line> lines;

170 for ( size_t i = 0; i < _points.size() - 1; ++i ) {

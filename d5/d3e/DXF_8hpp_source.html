147 // Check Exists
148 if ( ! std::filesystem::exists(_filepath) )
149 throw std::runtime_error("File does not exist at: " + _filepath.string());
150
151 // Check if valid file
152 if ( ! std::filesystem::is_regular_file(_filepath) )
153 throw std::runtime_error("File is not a regular file at: " + _filepath.string());
154
155 const char* fileCharArr = _filepath.string().c_str();
156 dxfRW dxfReader = dxfRW(fileCharArr);

158 dxfReader.read(&drw_reader, 1);
159

161 _objectives = drw_reader.getObjectives();
162 _obstacles = drw_reader.getObstacles();

493inline auto DRW_Reader::get_arc_as_polygon(DRW_Coord center, double radius, double start_angle,
494 double end_angle) -> std::vector<VIPRA::f3d>
495{
496 std::vector<VIPRA::f3d> points;
497 double center_x = center.x;
498 double center_y = center.y;
499
500 // Increments by steps to create points around a center at the specified radius.
501 // For a circle, start point would be 0.00 and end point would be 2*pi
502 // Steps are defined as the angle resulting from subdividing 2*pi
503 double step_size = (2 * M_PI) / NUM_SUBDIVISIONS;
504 // TODO: consider tolerance with end state. Could create two points very close together.
505 for ( double step = start_angle; step < end_angle; step += step_size ) {
506 double x = center_x + radius * cos(step);
507 double y = center_y + radius * sin(step);
508
509 points.emplace_back(x, y, 0);
510 }
511
512 // Should add point on last point if not circle. This closes
513 // the shape if the arc is attached to another shape
514 if ( abs(end_angle - (2 * M_PI)) > 0.01 ) {
515 points.emplace_back(center_x + (radius * cos(end_angle)), center_y + (radius * sin(end_angle)), 0);
516 }
517
518 return points;
519}

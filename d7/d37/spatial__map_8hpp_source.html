21 std::vector<VIPRA::f3d> const& newPositions)
22 {
23 // for all of the data, check if they moved grids, if they did update both
24 for ( VIPRA::idx currIdx = 0; currIdx < newPositions.size(); ++currIdx ) {
25 auto& oldGrid = get_grid(oldPositions[currIdx]);
26 auto& newGrid = get_grid(newPositions[currIdx]);
27
28 if ( &oldGrid == &newGrid ) continue;
29
30 oldGrid.erase(std::remove(oldGrid.begin(), oldGrid.end(), currIdx), oldGrid.end());
31 newGrid.push_back(currIdx);
32 }

78 VIPRA_PERF_FUNC void initialize_grids(std::vector<VIPRA::f3d> const& positions,
79 std::vector<data_t> const& data)
80 {
81 if ( positions.size() != data.size() )
82 throw std::logic_error("Spatial Map initialized with differing counts of data and positions");
83
84 // Create grids
85 for ( auto& grid : _grid ) {
86 grid = std::vector<data_t>();

89 // Add all of the pedestrians in each grid to their respective grid
90 for ( VIPRA::idx currIdx = 0; currIdx < positions.size(); ++currIdx ) {
91 auto& grid = get_grid(positions[currIdx]);
92 grid.push_back(data[currIdx]);

102 [[nodiscard]] VIPRA_INLINE auto get_grid(VIPRA::f3d pos) -> std::vector<VIPRA::idx>&
103 {
104 assert(! out_of_bounds(pos));
105
106 auto gridX = static_cast<size_t>(pos.x / _cellSize);
107 auto gridY = static_cast<size_t>(pos.y / _cellSize);
108
109 return _grid[gridX + gridY * _cols];
110 }

118 [[nodiscard]] VIPRA_INLINE auto get_grid(VIPRA::f3d pos) const -> std::vector<VIPRA::idx> const&
119 {
120 assert(! out_of_bounds(pos));
121
122 return _grid[static_cast<size_t>(pos.x / _cellSize) + static_cast<size_t>(pos.y / _cellSize) * _cols];
123 }

133 [[nodiscard]] VIPRA_INLINE auto out_of_bounds(VIPRA::f_pnt gridX, VIPRA::f_pnt gridY) const -> bool
134 {
135 return gridX < 0 || gridX >= _cols * _cellSize || gridY < 0 || gridY >= _rows * _cellSize;
136 }

102template <typename pedset_t, typename obstacles_t, typename goals_t>
103void HumanBehavior<pedset_t, obstacles_t, goals_t>::initialize(pedset_t const& pedset,
104 obstacles_t const& obstacles, goals_t& goals)
105{
106 VIPRA::State dummyState;
107
108 _context.engine = VIPRA::Random::Engine{_seedNum};
109 _context.pedStates = std::vector<Behaviors::stateUID>(pedset.num_pedestrians());
110 _context.types = std::vector<Behaviors::typeUID>(pedset.num_pedestrians());
111
112 _conditionMet.resize(pedset.num_pedestrians(), false);
113 _targets.resize(pedset.num_pedestrians());
114 for ( VIPRA::idx i = 0; i < pedset.num_pedestrians(); ++i ) {
115 _targets[i] = Target{TargetType::PEDESTRIAN, i};
116 }
117
118 Simpack<pedset_t, obstacles_t, goals_t> pack{pedset, obstacles, goals, _selector.get_groups(),
119 _context, dummyState, 0};
120 _selector.initialize(_name, pack);
121
122 for ( auto& actionGroup : _actions ) {
123 for ( auto& action : actionGroup ) {
124 action.initialize(pack);
125 }
126 }
127}

162template <typename pedset_t, typename obstacles_t, typename goals_t>
163void HumanBehavior<pedset_t, obstacles_t, goals_t>::apply_actions(pedset_t& pedset, obstacles_t& map,
164 goals_t& goals, VIPRA::State& state,
165 VIPRA::delta_t deltaT)
166{
167 GroupsContainer& groups = _selector.get_groups();
168 const VIPRA::size groupCnt = groups.size();
169 HumanBehavior<pedset_t, obstacles_t, goals_t>::pack_t pack{pedset, map, goals, _selector.get_groups(),
170 _context, state, deltaT};
171
172 // check if any of the pedestrians should have an action applied
173 for ( VIPRA::idx i = 0; i < groupCnt; ++i ) {
174 auto& pedestrians = groups[i];
175
176 // remove any pedestrians from the group if they have left the simulation
177 pedestrians.erase(std::remove_if(pedestrians.begin(), pedestrians.end(),
178 [&](VIPRA::idx ped) { return goals.is_goal_met(ped); }),
179 pedestrians.end());
180
181 // for each of the group's actions, check if it should apply and apply it
182 std::for_each(_actions[i].begin(), _actions[i].end(), [&](auto& action) {
183 if ( action.has_target() ) {
184 action.targets()->get_targets(pack, pedestrians, _targets);
185 // TODO(rolland): this needs to go back to self, if there is an action that changes it
186 }
187
188 if ( action.has_condition() ) {
189 // check condition and apply
190 action.condition()->evaluate(pack, pedestrians, _conditionMet, _targets, action.duration());
191 action.perform_action(pack, pedestrians, _conditionMet, _targets);
192 }
193 else {
194 // unconditional
195 action.perform_action(pack, pedestrians, _targets);
196 }
197 });
198 }

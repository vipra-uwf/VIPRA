101template <typename pedset_t, typename map_t, typename goals_t>
102void HumanBehavior<pedset_t, map_t, goals_t>::initialize(pedset_t const& pedset, map_t const& map,
103 goals_t& goals)
104{
105 VIPRA::State dummyState;
106
107 _context.engine = VIPRA::Random::Engine{_seedNum};
108 _context.pedStates = std::vector<Behaviors::stateUID>(pedset.num_pedestrians());
109 _context.types = std::vector<Behaviors::typeUID>(pedset.num_pedestrians());
110
111 _conditionMet.resize(pedset.num_pedestrians(), false);
112 _targets.resize(pedset.num_pedestrians());
113 for ( VIPRA::idx i = 0; i < pedset.num_pedestrians(); ++i ) {
114 _targets[i] = Target{TargetType::PEDESTRIAN, i};
115 }
116
117 Simpack<pedset_t, map_t, goals_t> pack{pedset, map, goals, _selector.get_groups(), _context, dummyState, 0};
118 _selector.initialize(_name, pack);
119
120 for ( auto& actionGroup : _actions ) {
121 for ( auto& action : actionGroup ) {
122 action.initialize(pack);
123 }
124 }
125}

159template <typename pedset_t, typename map_t, typename goals_t>
160void HumanBehavior<pedset_t, map_t, goals_t>::apply_actions(pedset_t& pedset, map_t& map, goals_t& goals,
161 VIPRA::State& state, VIPRA::delta_t deltaT)
162{
163 GroupsContainer& groups = _selector.get_groups();
164 const VIPRA::size groupCnt = groups.size();
165 HumanBehavior<pedset_t, map_t, goals_t>::pack_t pack{pedset, map, goals, _selector.get_groups(),
166 _context, state, deltaT};
167
168 // check if any of the pedestrians should have an action applied
169 for ( VIPRA::idx i = 0; i < groupCnt; ++i ) {
170 auto& pedestrians = groups[i];
171
172 // remove any pedestrians from the group if they have left the simulation
173 pedestrians.erase(std::remove_if(pedestrians.begin(), pedestrians.end(),
174 [&](VIPRA::idx ped) { return goals.is_goal_met(ped); }),
175 pedestrians.end());
176
177 // for each of the group's actions, check if it should apply and apply it
178 std::for_each(_actions[i].begin(), _actions[i].end(), [&](auto& action) {
179 if ( action.has_target() ) {
180 action.targets()->get_targets(pack, pedestrians, _targets);
181 // TODO(rolland): this needs to go back to self, if there is an action that changes it
182 }
183
184 if ( action.has_condition() ) {
185 // check condition and apply
186 action.condition()->evaluate(pack, pedestrians, _conditionMet, _targets, action.duration());
187 action.perform_action(pack, pedestrians, _conditionMet, _targets);
188 }
189 else {
190 // unconditional
191 action.perform_action(pack, pedestrians, _targets);
192 }
193 });
194 }

94 void set_end_goals(auto const& pedset, auto const& map, VIPRA::Random::Engine& engine)
95 {
96 assert(pedset.num_pedestrians() > 0);
97
98 const VIPRA::size pedCnt = pedset.num_pedestrians();
99
100 // find the end goals, provided as a module parameter
101 auto const& objects = map.get_objectives(_endGoalType);
102 if ( objects.empty() ) {
103 throw std::runtime_error("No objects of type " + _endGoalType + " found in map");
104 }
105
106 // set each pedestrians end goal as the nearest of that type
107 for ( VIPRA::idx pedIdx = 0; pedIdx < pedCnt; ++pedIdx ) {
108 auto const pos = pedset.ped_coords(pedIdx);
109
110 auto const nearestGoalIter = get_nearest_goal(pos, objects);
111 VIPRA::Log::debug("Ped {} Nearest End Goal: ({}, {})", pedIdx, (*nearestGoalIter).center().x,
112 (*nearestGoalIter).center().y);
113
114 if ( nearestGoalIter != objects.end() ) {
115 const auto point = (*nearestGoalIter).random_point(engine);
116 VIPRA::Log::debug("Ped {} Final End Goal: ({}, {})", pedIdx, point.x, point.y);
117 set_end_goal(pedIdx, point);
118 }
119 else {
120 throw std::runtime_error("No goal found for pedestrian");
121 }
122 }

131 [[nodiscard]] static auto squash_path(std::vector<VIPRA::f3d> const& path) -> std::vector<VIPRA::f3d>
132 {
133 assert(path.empty() == false);
134
135 std::vector<VIPRA::f3d> squashedPath;
136
137 VIPRA::f3d dif;
138 for ( VIPRA::idx i = 1; i < path.size(); ++i ) {
139 auto currDif = path[i] - path[i - 1];
140 if ( currDif != dif ) {
141 squashedPath.push_back(path[i - 1]);
142 dif = currDif;
143 }
144 }
145
146 return squashedPath;
147 }

156 [[nodiscard]] static auto get_nearest_goal(VIPRA::f3d pos,
157 std::vector<VIPRA::Geometry::Polygon> const& goals)
158 -> std::vector<VIPRA::Geometry::Polygon>::const_iterator
159 {
160 assert(goals.empty() == false);
161
162 return std::min_element(goals.begin(), goals.end(), [&](auto const& left, auto const& right) {
163 return pos.distance_to(left.center()) < pos.distance_to(right.center());
164 });

173 void find_path(VIPRA::idx pedIdx, VIPRA::f3d startPos)
174 {
175 // Get the pedestrian start and end grid location
176 VIPRA::idx startIdx = _graph.get_closest_grid_idx(startPos);
177 VIPRA::idx endIdx = _graph.get_closest_grid_idx(end_goal(pedIdx));
178
179 if ( _graph.nodes().size() <= startIdx || _graph.nodes().size() <= endIdx ) {
180 throw std::runtime_error("Start or end index is out of bounds");
181 }
182
183 // use A* to find the path
184 auto const path = VIPRA::Algo::astar(
185 startIdx, endIdx, _graph,
186 [&](VIPRA::idx left, VIPRA::idx right) -> VIPRA::f_pnt {
187 // distance comparison function
188 return _graph.data(left).pos.distance_to(_graph.data(right).pos);
189 },
190 [&](VIPRA::idx nodeIdx) -> VIPRA::f3d { return _graph.data(nodeIdx).pos; });
191
192 if ( ! path ) {
193 // No path found
194 throw std::runtime_error("No path found for pedestrian " + std::to_string(pedIdx));
195 }
196
197 // set their path, squash nodes that all go in the same direction into one node
198 _paths[pedIdx] = squash_path(*path);
199 }

223 [[nodiscard]] inline static auto is_coord_loc_compare(CAttributeValue value1, CAttributeValue value2)
224 -> bool

226 return (value1.type == Type::COORD && value2.type == Type::LOCATION) ||

239 [[nodiscard]] inline static auto coord_loc_compare(CAttributeValue value1, CAttributeValue value2,
240 auto pack) -> bool
241 {
242 if ( value1.type == Type::COORD && value2.type == Type::LOCATION ) {
243 return pack.context.locations[value2.as<VIPRA::idx>()].contains(value1.as<VIPRA::f3d>());
244 }
245
246 if ( value2.type == Type::COORD && value1.type == Type::LOCATION ) {
247 return pack.context.locations[value1.as<VIPRA::idx>()].contains(value2.as<VIPRA::f3d>());
248 }
249
250 return false;
251 }

261 [[nodiscard]] inline static auto is_equal(CAttributeValue value1, CAttributeValue value2, auto pack) -> bool
262 {
263 if ( is_coord_loc_compare(value1, value2) ) {
264 return coord_loc_compare(value1, value2, pack);
265 }
266
267 if ( value1.type != value2.type ) return false;
268
269 switch ( value1.type ) {
270 case Type::INVALID:
271 throw std::runtime_error("Invalid Attribute Type");
272 return false;
273 case Type::NUMBER:
274 return value1.as<VIPRA::f_pnt>() == value2.as<VIPRA::f_pnt>();
275 case Type::COORD:
276 return value1.as<VIPRA::f3d>() == value2.as<VIPRA::f3d>();
277 case Type::STATE:
278 return value1.as<stateUID>() == value2.as<stateUID>();
279 case Type::STATUS:
280 return value1.as<EventStatus>() == value2.as<EventStatus>();
281 case Type::LOCATION:
282 return value1.as<VIPRA::idx>() == value2.as<VIPRA::idx>();
283 break;
284 default:
285 break;
286 }
287
288 return false;
289 }

321 inline static void cleanup()
322 {
323 for ( auto data : get_value_store() ) {
324 // NOLINTBEGIN(cppcoreguidelines-owning-memory)
325 switch ( data.type ) {
326 case Type::INVALID:
327 // throw std::runtime_error("Invalid Type");
328 break;
329 case Type::NUMBER:
330 delete data.as_ptr<VIPRA::f_pnt>();
331 break;
332 case Type::COORD:
333 delete data.as_ptr<VIPRA::f3d>();
334 break;
335 case Type::STATE:
336 delete data.as_ptr<stateUID>();
337 break;
338 case Type::STATUS:
339 delete data.as_ptr<EventStatus>();
340 break;
341 case Type::LOCATION:
342 [[fallthrough]];
343 case Type::TOWARDS_LOC:
344 delete data.as_ptr<VIPRA::idx>();
345 break;
346 case Type::TOWARDS_ATTR:
347 delete data.as_ptr<Attribute>();
348 break;
349 default:
350 // throw std::runtime_error("Cleanup Error: Unknown Type");
351 break;
352 }
353 // NOLINTEND(cppcoreguidelines-owning-memory)
354 }
355
356 get_value_store().clear();
357 }

594 static inline void set_goal(Target target, auto pack, CAttributeValue value)
595 {
596 auto& goals = pack.goals;
597 auto& context = pack.context;
598 auto const& pedset = pack.pedset;
599
600 if ( value.type == Type::COORD ) {
601 goals.change_end_goal(target.targetIdx, pedset.ped_coords(target.targetIdx), value.as<VIPRA::f3d>());
602 }
603 else if ( value.type == Type::LOCATION ) {
604 // TODO(rolland): this doesn't take into account two pedestrains going to the same location
605 goals.change_end_goal(target.targetIdx, pedset.ped_coords(target.targetIdx),
606 context.locations[value.as<VIPRA::idx>()].random_point(context.engine));
607 }

609 value.type_check(Type::COORD);

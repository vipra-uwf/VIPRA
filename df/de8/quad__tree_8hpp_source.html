70 [[nodiscard]] auto ray_hit(VIPRA::f3d start, VIPRA::f3d end) const -> VIPRA::f_pnt {
71 VIPRA::f_pnt hit = std::numeric_limits<VIPRA::f_pnt>::max();
72 for (auto const& obstacle : _obstacles) {
73 for (auto const& edge : obstacle.edges) {
74 if (!edge.does_intersect({start, end})) continue;
75 auto intersection = edge.intersection_point({start, end});
76 hit = std::min(hit, start.distance_to(intersection));
77 }

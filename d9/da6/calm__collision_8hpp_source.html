80 {
81 auto const& coords = pedset.all_coords();
82 auto const& velocities = pedset.all_velocities();
83 auto const& shldrs = data.shoulderLens;
84 const VIPRA::size pedCnt = pedset.num_pedestrians();
85
86 for ( VIPRA::idx i = 0; i < pedCnt; ++i ) {
87 VIPRA::f3d pedVel = velocities[i];
88
89 if ( pedVel == VIPRA::f3d{} ) {
90 auto goalCoords = goals.current_goal(i);
91 auto goalDirection = goalCoords - coords[i];
92
93 auto shldr1 = (VIPRA::f3d{-goalDirection.y, goalDirection.x}.unit() * shldrs[i]) + coords[i];
94 auto shldr2 = (VIPRA::f3d{goalDirection.y, -goalDirection.x}.unit() * shldrs[i]) + coords[i];
95
96 const VIPRA::f3d range = (goalDirection.unit() * RECTANGLE_RANGE);
97 collisionRectangles[i] = Rectangle{shldr1, shldr1 + range, shldr2 + range, shldr2};
98 continue;
99 }
100
101 const Line shoulders = get_shoulder_points(coords[i], pedVel, shldrs[i]);
102 const VIPRA::f3d range = (pedVel.unit() * RECTANGLE_RANGE);
103 collisionRectangles[i] =

108 VIPRA_PERF_FUNC static void add_intersection_points(VIPRA::f3d start1, VIPRA::f3d end1, VIPRA::f3d start2,
109 VIPRA::f3d end2, VIPRA::f3dVec& intersectionPoints)
110 {
111 // NOLINTBEGIN(readability-identifier-length)
112 VIPRA::f_pnt a1 = end1.y - start1.y;
113 VIPRA::f_pnt b1 = start1.x - end1.x;
114 VIPRA::f_pnt a2 = end2.y - start2.y;
115 VIPRA::f_pnt b2 = start2.x - end2.x;
116 VIPRA::f_pnt c1 = a1 * start1.x + b1 * start1.y;
117 VIPRA::f_pnt c2 = a2 * start2.x + b2 * start2.y;
118 // NOLINTEND(readability-identifier-length)
119
120 VIPRA::f_pnt det = a1 * b2 - a2 * b1;
121
122 if ( det == 0 ) {
123 if ( Line{start1, end1}.is_point_on(start2) ) intersectionPoints.push_back(start2);
124 if ( Line{start1, end1}.is_point_on(end2) ) intersectionPoints.push_back(end2);
125 if ( Line{start2, end2}.is_point_on(start1) ) intersectionPoints.push_back(start1);
126 if ( Line{start2, end2}.is_point_on(end1) ) intersectionPoints.push_back(start1);
127 }
128 else {
129 VIPRA::f_pnt pointX{};
130 VIPRA::f_pnt pointY{};
131 pointX = (b2 * c1 - b1 * c2) / det;
132 pointY = -(a2 * c1 - a1 * c2) / det;
133 intersectionPoints.emplace_back(pointX, pointY);
134 }
135 }

137 [[nodiscard]] VIPRA_PERF_FUNC auto get_collision_area_midpoint(VIPRA::idx index1,
138 VIPRA::idx index2) -> VIPRA::f3d
139 {
140 Rectangle const& rec1 = collisionRectangles[index1];
141 Rectangle const& rec2 = collisionRectangles[index2];
142
143 auto const& sides1 = rec1.sides();
144 auto const& sides2 = rec2.sides();
145
146 VIPRA::f3dVec intersectionPoints;
147 for ( size_t i = 0; i < 4; i++ ) {
148 for ( size_t j = 0; j < 4; j++ ) {
149 if ( sides1[i].does_intersect(sides2[j]) ) {
150 add_intersection_points(sides1[i].start, sides1[i].end, sides2[j].start, sides2[j].end,
151 intersectionPoints);
152 }
153 }
154 }
155
156 VIPRA::f3d midpoint(0, 0, 0);
157 for ( auto const& intersectionPoint : intersectionPoints ) {
158 midpoint += intersectionPoint;
159 }
160 midpoint /= intersectionPoints.size();
161
162 return midpoint;
163 }

165 [[nodiscard]] VIPRA_PERF_FUNC auto check_if_highest_priority(auto const& pedset, auto const& goals,
166 VIPRA::idx pedIdx,
167 VIPRA::timestep /*timestep*/) -> bool
168 {
169 auto neighbors = pedset.all_neighbors_within(pedIdx, 1);
170
171 bool flag = true;
172 for ( VIPRA::idx otherIdx : neighbors ) {
173 if ( otherIdx == pedIdx ) continue;
174 if ( goals.is_goal_met(otherIdx) ) continue;
175
176 auto coords1 = pedset.ped_coords(pedIdx);
177 auto coords2 = pedset.ped_coords(otherIdx);
178
179 constexpr VIPRA::f_pnt DIST2 = (4 * RECTANGLE_RANGE * RECTANGLE_RANGE);
180 if ( coords1.distance_to_sqrd(coords2) >= DIST2 ) continue;
181
182 bool cr2in1 = collisionRectangles[pedIdx].is_point_inside(coords2);
183 bool cr1in2 = collisionRectangles[otherIdx].is_point_inside(coords1);
184
185 auto goal1 = goals.current_goal(pedIdx);
186 auto goal2 = goals.current_goal(otherIdx);
187
188 // If goals don't match, check if coordinates of the pedestrians are in
189 // each others collision rectangles
190 if ( goal1 != goal2 ) {
191 if ( ! cr2in1 ) {
192 inRace[pedIdx][otherIdx] = false;
193 inRace[otherIdx][pedIdx] = false;
194 continue;
195 }
196
197 if ( ! cr1in2 ) {
198 inRace[otherIdx][pedIdx] = true;
199 inRace[pedIdx][otherIdx] = true;
200 flag = false;
201 continue;
202 }
203 }
204
205 // Check if pedestrians collide, if not continue onto next passenger
206 bool collisionCheck = check_if_collide(pedIdx, otherIdx);
207 if ( ! collisionCheck ) {
208 if ( inRace[pedIdx][otherIdx] ) {
209 inRace[pedIdx][otherIdx] = false;
210 inRace[otherIdx][pedIdx] = false;
211 }
212 continue;
213 }
214
215 // If goals match, use distance comparision
216 if ( goal1 == goal2 ) {
217 if ( coords1.distance_to_sqrd(goal1) > coords2.distance_to_sqrd(goal2) ) {
218 inRace[otherIdx][pedIdx] = true;
219 inRace[pedIdx][otherIdx] = true;
220 flag = false;
221 }
222 else {
223 if ( inRace[pedIdx][otherIdx] ) {
224 inRace[pedIdx][otherIdx] = false;
225 inRace[otherIdx][pedIdx] = false;
226 }
227 }
228 continue;
229 }
230
231 // If goals don't match, use collision midpoint to resolve race condition
232 VIPRA::f3d collisionMidpoint;
233 if ( ! inRace[pedIdx][otherIdx] ) {
234 collisionMidpoint = get_collision_area_midpoint(pedIdx, otherIdx);
235
236 velocityDirections[pedIdx] = pedset.ped_velocity(pedIdx).unit();
237
238 intersectionMidpoints[pedIdx][otherIdx] = collisionMidpoint;
239 intersectionMidpoints[otherIdx][pedIdx] = collisionMidpoint;
240
241 inRace[otherIdx][pedIdx] = true;
242 inRace[pedIdx][otherIdx] = true;
243 }
244 else {
245 collisionMidpoint = intersectionMidpoints[pedIdx][otherIdx];
246 }
247 if ( coords1.distance_to_sqrd(collisionMidpoint) > coords2.distance_to_sqrd(collisionMidpoint) )
248 flag = false;
249 else if ( coords1.distance_to_sqrd(collisionMidpoint) == coords2.distance_to_sqrd(collisionMidpoint) ) {
250 if ( pedIdx < otherIdx ) flag = false;
251 }
252 }
253 return flag;
254 }

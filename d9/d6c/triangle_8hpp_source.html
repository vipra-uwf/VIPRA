40{
41 auto const sign = [](f3d point1, f3d point2, f3d point3) -> f_pnt {
42 return (point1.x - point3.x) * (point2.y - point3.y) -
43 (point2.x - point3.x) * (point1.y - point3.y);
44 };
45
46 const f_pnt dir1 = sign(point, _points[0], _points[1]);
47 const f_pnt dir2 = sign(point, _points[1], _points[2]);
48 const f_pnt dir3 = sign(point, _points[2], _points[0]);
49
50 return ((dir1 >= 0) && (dir2 >= 0) && (dir3 >= 0)) ||
51 ((dir1 <= 0) && (dir2 <= 0) && (dir3 <= 0));
52}

260 template <typename comp_t>
261 [[nodiscard]] static auto find_event_component(
262 BehaviorParser::EventContext* ctx) -> std::optional<comp_t>
263 {
264 for ( auto const& attr : ctx->event_attribute() ) {
265 // Name
266 if constexpr ( std::is_same_v<comp_t, std::string> )
267 if ( attr->event_name() ) return attr->event_name()->ID()->toString();
268
269 // Start Condition
270 if constexpr ( std::is_same_v<comp_t, evStart> )
271 if ( attr->event_start() ) return attr->event_start();
272
273 // End Condition
274 if constexpr ( std::is_same_v<comp_t, evEnd> )
275 if ( attr->event_end() ) return attr->event_end();
276 }
277
278 return std::nullopt;
279 }

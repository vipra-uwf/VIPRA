240 template <typename comp_t>
241 [[nodiscard]] static auto find_event_component(BehaviorParser::EventContext* ctx) -> std::optional<comp_t>
242 {
243 for ( auto const& attr : ctx->event_attribute() ) {
244 // Name
245 if constexpr ( std::is_same_v<comp_t, std::string> )
246 if ( attr->event_name() ) return attr->event_name()->ID()->toString();
247
248 // Start Condition
249 if constexpr ( std::is_same_v<comp_t, evStart> )
250 if ( attr->event_start() ) return attr->event_start();
251
252 // End Condition
253 if constexpr ( std::is_same_v<comp_t, evEnd> )
254 if ( attr->event_end() ) return attr->event_end();
255 }
256
257 return std::nullopt;
258 }
259
267 template <typename comp_t>
268 [[nodiscard]] static auto find_action_component(BehaviorParser::ActionContext* ctx) -> std::optional<comp_t>
269 {
270 for ( auto const& attr : ctx->action_attribute() ) {
271 // Stimulus
272 if constexpr ( std::is_same_v<comp_t, acStimulus> )
273 if ( attr->action_stimulus() ) return attr->action_stimulus();
274
275 // Response
276 if constexpr ( std::is_same_v<comp_t, acResponse> )
277 if ( attr->action_response() ) return attr->action_response();
278
279 // Target
280 if constexpr ( std::is_same_v<comp_t, acTarget> )
281 if ( attr->action_target() ) return attr->action_target();
282
283 // Duration
284 if constexpr ( std::is_same_v<comp_t, acDuration> )
285 if ( attr->action_duration() ) return attr->action_duration();
286 }

288 return std::nullopt;

358template <typename pedset_t, typename map_t, typename goals_t>
359auto BehaviorBuilder<pedset_t, map_t, goals_t>::build(std::string behaviorName,
360 std::filesystem::path const& filepath,
361 seed seedNum) -> HumanBehavior
362{
363 if ( ! std::filesystem::exists(filepath) ) {
364 VIPRA::Log::error("Behavior \"{}\" Does NOT Exist at {}", behaviorName, filepath.c_str());
365 BuilderException::error();
366 }
367
368 VIPRA::Log::debug("Loading Behavior: {} at {}", behaviorName, std::filesystem::canonical(filepath).c_str());
369 initial_behavior_setup(behaviorName, seedNum);
370
371 std::ifstream dslFile(filepath);
372
373 antlr4::ANTLRInputStream input(dslFile);
374 BehaviorLexer lexer(&input);
375 antlr4::CommonTokenStream tokens(&lexer);
376 BehaviorParser parser(&tokens);
377
378 parser.removeErrorListeners();
379 parser.addErrorListener(&_errorListener);
380
381 BehaviorParser::ProgramContext* tree = parser.program();
382
383 visitProgram(tree);
384
385 end_behavior_check();
386
387 return _currentBehavior;
388}

679 BuilderException::error();

1098template <typename pedset_t, typename map_t, typename goals_t>
1099auto BehaviorBuilder<pedset_t, map_t, goals_t>::visitEvent(BehaviorParser::EventContext* ctx) -> antlrcpp::Any
1100{
1101 auto name = find_event_component<evName>(ctx);
1102 if ( ! name ) error(R"(Behavior "{}": Missing Event Name)", _currentBehavior.get_name());
1103 std::string eventName = "!" + name.value();
1104
1105 if ( get_event(eventName) )
1106 error(R"(Behavior "{}": Attempt To Redefine Event: "{}")", _currentBehavior.get_name(), eventName);
1107
1108 auto startCond = find_event_component<evStart>(ctx);
1109 if ( ! startCond ) error(R"(Behavior "{}": Missing Start Condition)", _currentBehavior.get_name());
1110
1111 auto endCond = find_event_component<evEnd>(ctx);
1112
1113 // NOTE(rolland): a bit odd but we need to ensure that the event is in the map, incase the conditions rely on it
1114 _eventsMap[eventName] = _currentBehavior.event_count();
1115
1116 Event event(eventName);
1117
1118 VIPRA::Log::debug(R"(Behavior "{}": Event: "{}" Adding Start Condition)", _currentBehavior.get_name(),
1119 eventName);
1120 event.set_start_condition(build_condition(startCond.value()->condition()));
1121
1122 if ( endCond ) {
1123 VIPRA::Log::debug(R"(Behavior "{}": Event: "{}" Adding End Condition)", _currentBehavior.get_name(),
1124 eventName);
1125 event.set_end_condition(build_condition(endCond.value()->condition()));
1126 }
1127
1128 _eventsMap[eventName] = _currentBehavior.add_event(event);
1129
1130 return BehaviorBaseVisitor::visitEvent(ctx);
1131}
1132template <typename pedset_t, typename map_t, typename goals_t>
1133auto BehaviorBuilder<pedset_t, map_t, goals_t>::visitAction(BehaviorParser::ActionContext* ctx)
1134 -> antlrcpp::Any
1135{
1136 auto response = find_action_component<acResponse>(ctx);
1137 if ( ! response ) error(R"(Behavior "{}": Action has no Response)", _currentBehavior.get_name());
1138
1139 auto stimulus = find_action_component<acStimulus>(ctx);
1140 auto target = find_action_component<acTarget>(ctx);
1141 auto duration = find_action_component<acDuration>(ctx);
1142
1143 Action action;
1144
1145 auto const typeStr = ctx->ID()->toString();
1146 auto const type = get_check_type(typeStr);
1147
1148 VIPRA::Log::debug("Behavior \"{}\": Adding Action For {}", _currentBehavior.get_name(), typeStr);
1149
1150 auto atoms = response.value()->sub_action()->action_atom();
1151 std::for_each(atoms.begin(), atoms.end(),
1152 [&](BehaviorParser::Action_atomContext* atom) { add_atom_to_action(action, atom); });
1153
1154 if ( stimulus ) action.add_condition(build_condition(stimulus.value()->condition()));
1155 if ( target ) add_target_to_action(action, target.value()->target());
1156 if ( duration ) action.add_duration(get_numeric(duration.value()->duration()->value_numeric(), _currSeed));
1157
1158 _currentBehavior.add_action(type, action);
1159 return BehaviorBaseVisitor::visitAction(ctx);
1160}

158 void init_random_peds(auto const& map, VIPRA::Random::Engine& engine)
159 {
160 // TODO(rolland, issue #50): this is terrible
161
162 assert(_coords.size() == _velocities.size());
163
164 auto spawnAreas = map.get_spawns();
165 if ( spawnAreas.size() == 0 ) {
166 VIPRA::Log::warn(
167 "{} Pedestrians were set to spawn randomly, but no spawn areas were "
168 "provided in the map",
169 _randomPedCnt);
170 return;
171 }
172
173 size_t startSize = _coords.size();
174 _coords.resize(_coords.size() + _randomPedCnt);
175 _velocities.resize(_velocities.size() + _randomPedCnt);
176
177 VIPRA::Random::uniform_distribution<size_t> polyDist{0,
178 spawnAreas.size() - 1};
179
180 // go through each pedestrian and give them a random point inside one of the spawn points
181 for ( size_t i = startSize; i < _coords.size(); ++i ) {
182 // TODO(rolland, issue #50): effectively infinite loop if the spawn is covered by an obstacle
183 VIPRA::idx spawnIdx = polyDist(engine);
184
185 do {
186 _coords[i] = spawnAreas[spawnIdx].random_point(engine);
187 VIPRA::Log::debug("Random Point: ({}, {}) Collides: {}", _coords[i].x,
188 _coords[i].y, map.collision(_coords[i]));
189 } while ( map.collision(Geometry::Circle{_coords[i], 0.4}) );
190 }
191
192 VIPRA::Log::debug("{} pedestrians added randomly to {} spawn areas",
193 _randomPedCnt, spawnAreas.size());
194 }

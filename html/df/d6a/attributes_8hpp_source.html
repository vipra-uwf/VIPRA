214 [[nodiscard]] inline static auto is_coord_loc_compare(CAttributeValue value1, CAttributeValue value2)
215 -> bool {
216 return (value1.type == Type::COORD && value2.type == Type::LOCATION) ||
217 (value1.type == Type::LOCATION && value2.type == Type::COORD);
218 }

229 [[nodiscard]] inline static auto coord_loc_compare(CAttributeValue value1, CAttributeValue value2,
230 auto pack) -> bool {
231 if (value1.type == Type::COORD && value2.type == Type::LOCATION) {
232 return pack.context.locations[value2.as<VIPRA::idx>()].contains(value1.as<VIPRA::f3d>());
233 }
234
235 if (value2.type == Type::COORD && value1.type == Type::LOCATION) {
236 return pack.context.locations[value1.as<VIPRA::idx>()].contains(value2.as<VIPRA::f3d>());
237 }
238
239 return false;
240 }

250 [[nodiscard]] inline static auto is_equal(CAttributeValue value1, CAttributeValue value2, auto pack)
251 -> bool {
252 if (is_coord_loc_compare(value1, value2)) {
253 return coord_loc_compare(value1, value2, pack);
254 }
255
256 if (value1.type != value2.type) return false;
257
258 switch (value1.type) {
259 case Type::INVALID:
260 throw std::runtime_error("Invalid Type");
261 return false;
262 case Type::NUMBER:
263 return value1.as<VIPRA::f_pnt>() == value2.as<VIPRA::f_pnt>();
264 case Type::COORD:
265 return value1.as<VIPRA::f3d>() == value2.as<VIPRA::f3d>();
266 case Type::STATE:
267 return value1.as<stateUID>() == value2.as<stateUID>();
268 case Type::STATUS:
269 return value1.as<EventStatus>() == value2.as<EventStatus>();
270 case Type::LOCATION:
271 return value1.as<VIPRA::idx>() == value2.as<VIPRA::idx>();
272 break;
273 default:
274 break;
275 }
276
277 return false;
278 }

308 inline static void cleanup() {
309 for (auto data : get_value_store()) {
310 // NOLINTBEGIN(cppcoreguidelines-owning-memory)
311 switch (data.type) {
312 case Type::INVALID:
313 // throw std::runtime_error("Invalid Type");
314 break;
315 case Type::NUMBER:
316 delete data.as_ptr<VIPRA::f_pnt>();
317 break;
318 case Type::COORD:
319 delete data.as_ptr<VIPRA::f3d>();
320 break;
321 case Type::STATE:
322 delete data.as_ptr<stateUID>();
323 break;
324 case Type::STATUS:
325 delete data.as_ptr<EventStatus>();
326 break;
327 case Type::LOCATION:
328 [[fallthrough]];
329 case Type::TOWARDS_LOC:
330 delete data.as_ptr<VIPRA::idx>();
331 break;
332 case Type::TOWARDS_ATTR:
333 delete data.as_ptr<Attribute>();
334 break;
335 default:
336 // throw std::runtime_error("Cleanup Error: Unknown Type");
337 break;
338 }
339 // NOLINTEND(cppcoreguidelines-owning-memory)
340 }
341
342 get_value_store().clear();
343 }

571 static inline void set_goal(Target target, auto pack, CAttributeValue value) {
572 auto& goals = pack.goals;
573 auto& context = pack.context;
574 auto const& pedset = pack.pedset;
575
576 if (value.type == Type::COORD) {
577 goals.change_end_goal(target.targetIdx, value.as<VIPRA::f3d>());
578 } else if (value.type == Type::LOCATION) {
579 // TODO(rolland): this doesn't take into account two pedestrains going to the same location
580 goals.change_end_goal(target.targetIdx,
581 context.locations[value.as<VIPRA::idx>()].random_point(context.engine));
582 }
583
584 value.type_check(Type::COORD);
585 }

595 static inline void set_velocity(Target target, auto pack, VIPRA::State& state, CAttributeValue value) {
596 value.type_check(Type::COORD);
597
598 state.velocities[target.targetIdx] = value.as<VIPRA::f3d>();
599
600 VIPRA::f3d originalPos = pack.pedset.ped_coords(target.targetIdx);
601 state.positions[target.targetIdx] = originalPos + (state.velocities[target.targetIdx] * pack.dT);

641 value.type_check(Type::NUMBER);

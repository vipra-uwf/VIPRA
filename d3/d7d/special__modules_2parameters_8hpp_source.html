153template <typename input_t>
154template <typename data_t>
155auto Parameters<input_t>::get_param(Modules::Type module, std::string const& moduleName,
156 std::string const& paramName,
157 Random::Engine& engine) const -> std::remove_cvref_t<data_t>
158{
159 using param_t = std::remove_cvref_t<data_t>;
160
161 std::string moduleStr = to_string(module);
162
163 // Check that the parameter was registered
164 if ( ! contains(module, moduleName, paramName) ) {
165 throw std::runtime_error("Parameter: " + paramName + " For " + to_string(module) +
166 " Module: " + moduleName + " Not Registered");
167 }
168
169 // load the value of the parameter
170 auto value = std::optional<param_t>{};
171 if constexpr ( Util::is_specialization<param_t, std::vector>::value ) {
172 // the parameter is supposed to be an array, load it as such
173 value = get_array_param<param_t>(module, moduleName, paramName);
174 }
175 else {
176 // the parameter is meant to be a single value, follow parameter randomization rules
177 value = _input.template get_param<param_t>(moduleStr, moduleName, paramName, engine);
178 }
179
180 if ( ! value.has_value() ) {
181 // parameter value wasn't found, throw
182 throw std::runtime_error("Required Parameter: " + paramName + " For " + to_string(module) +
183 " Module: " + moduleName + " Not Provided In Input");
184 }
185
186 record_param(module, moduleName, paramName, value.value());
187
188 return param_t{std::move(value.value())};
189}

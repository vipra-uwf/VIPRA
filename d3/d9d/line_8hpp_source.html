109 [[nodiscard]] constexpr auto intersection_point(Line other) const noexcept
110 -> VIPRA::f3d
111 {
112 assert(does_intersect(other));
113
114 const VIPRA::f_pnt a1 = end.y - start.y;
115 const VIPRA::f_pnt b1 = start.x - end.x;
116 const VIPRA::f_pnt c1 = a1 * start.x + b1 * start.y;
117
118 const VIPRA::f_pnt a2 = other.end.y - other.start.y;
119 const VIPRA::f_pnt b2 = other.start.x - other.end.x;
120 const VIPRA::f_pnt c2 = a2 * other.start.x + b2 * other.start.y;
121
122 const VIPRA::f_pnt determinant = a1 * b2 - a2 * b1;
123
124 if ( determinant == 0 ) return VIPRA::f3d{0, 0, 0};
125
126 const VIPRA::f_pnt x = (b2 * c1 - b1 * c2) / determinant;
127 const VIPRA::f_pnt y = (a1 * c2 - a2 * c1) / determinant;
128
129 return VIPRA::f3d{x, y, 0};
130 }

140 [[nodiscard]] static constexpr auto do_intersect(Line line1,
141 Line line2) noexcept -> bool
142 {
143 const Orientation ori1 = orientation_to(line1, line2.start);
144 const Orientation ori2 = orientation_to(line1, line2.end);
145 const Orientation ori3 = orientation_to(line2, line1.start);
146 const Orientation ori4 = orientation_to(line2, line1.end);
147
148 if ( ori1 != ori2 && ori3 != ori4 ) return true;

150 if ( ori1 == Orientation::COLLINEAR && line1.is_point_on(line2.start) )
151 return true;

153 return true;

155 return true;

157 return true;
